#include <iostream>

using namespace std;

class NumeroA
{  int a;
   public:
     NumeroA(int);
     void setNumeroA(int);
     int getNumeroA(void);
     void operator += (NumeroA);     
};

NumeroA::NumeroA(int x)
{    a = x; 
}

void NumeroA::setNumeroA(int x)
{    this -> a = x; 
}

int NumeroA::getNumeroA(void)
{    return a; 
}

void NumeroA::operator += (NumeroA num)
{    this -> setNumeroA(num.getNumeroA()*(num.getNumeroA()-1)); 
}
